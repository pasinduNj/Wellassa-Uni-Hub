CREATE DATABASE IF NOT EXISTS wellassaunihub;

USE wellassaunihub;

CREATE TABLE admins (
    admin_id VARCHAR(10) PRIMARY KEY,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    contact_number VARCHAR(15) NOT NULL UNIQUE,
    joined_date DATE NOT NULL
);

CREATE TABLE customers (
    cus_id VARCHAR(10) PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    contact_number VARCHAR(15) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    joined_date DATE NOT NULL
);

CREATE TABLE service_providers (
    sp_id VARCHAR(10) PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    business_name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    nic VARCHAR(15) NOT NULL UNIQUE,
    contact_number VARCHAR(15) NOT NULL UNIQUE,
    whatsapp_number VARCHAR(15) NOT NULL UNIQUE,
    service_address VARCHAR(255) NOT NULL,
    service_type ENUM('Reservation Provider', 'Product Seller', 'Freelancer') NOT NULL,
    password VARCHAR(255) NOT NULL,
    joined_date DATE NOT NULL
);
